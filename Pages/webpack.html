What is Webpack

  • Module Bundler
  • Custom files or NPM Installed
  • Generates Static Assets
  • Extend with Plugins & Loaders
  • 각종 필요한 Depandency module 들을 하나의 파일로 묶어줌

Process

  • Webpack 모듈 설치
  • app.js 작성
  • 번들생성 - webpack app.js bunble.js
  • index.html 에서 import
Sample codes 
        
            npm install -g Webpack

            // To user the dev server
            npm install Webpack-dev-server

            // Bundling 
            webpack app.js bundle.js

            // Watch mode
            webpack app.js bundle.js --watch

            // install jQuery
            npm install jquery --save
            let $ = require('jquery');

            // Loaders
            npm install css-loader style-loader --save-dev

            // CSS 가 JavaScript 파일에 include
            require('!style-loader!css-loader!./style.css');

            // 아니면
            // webpack.config.js
            module.exports = {
                entiry: './src/js/app.js',
                output: {
                    path: __dirname + '/dist',
                    filename: 'bundle.js'
                },
                module: {
                    loaders:[
                        {test: /\.css$/, loader: "style-loader!css-loader"},
                        {test: /\.js$/, loader: 'babel-loader', exclude: /node_modules/, query:{presets:[es2015]}}
                    ]
                }
            }

            require('../css/style.css');

            // 터미널 webpack 실행
            webpack

            // OR package.json 의 scripts 에서
            "build" : "webpack"

            "start" : "webpack-dev-server --entry ./src/js/app.js --output-filename ./dist/bundle.js"
            
            npm install babel-core babel-loader babel-preset-es2015 --save-dev
